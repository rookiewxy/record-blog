```less

.vector3 {
  .xt-inputNumber {

    .xt-inputNumber-input {
      min-width: 38px;
      color: @xt-white-color-1;
      padding: 0 4px;
    }
  }

// 控制台为.xt-inputNumber .xt-inputNumber-input
.vector3 {
  .xt-inputNumber {

    &-input {
      min-width: 38px;
      color: @xt-white-color-1;
      padding: 0 4px;
    }
  }
}

```
