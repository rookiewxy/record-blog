                            localIdentName: '[local]__[hash:base64:5]' // 定义类名

                        loader: "less-loader",
                    },
                    // 在这里引入要增加的全局less文件

                        loader: 'style-resources-loader',
                        options: {
                            patterns: path.resolve(__dirname, './src/styles/index.less')
                        }
                    }
                ],
            }
